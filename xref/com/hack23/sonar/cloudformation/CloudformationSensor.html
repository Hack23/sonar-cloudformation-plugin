55  	public static final String SENSOR_NAME = "Cloudformation Check";

76  	 * Instantiates a new cloudformation sensor.

78  	 * @param configuration the configuration
79  	 * @param fileSystem    the file system
80  	 * @param pathResolver  the path resolver
81  	 */
82  	public CloudformationSensor(final CloudformationSensorConfiguration configuration, final FileSystem fileSystem,
83  			final PathResolver pathResolver) {
84  		super();
85  		this.cfnNagReportReader = new CfnNagReportReader();
86  		this.cfnNagScanReportReader = new CfnNagScanReportReader();
87  		this.configuration = configuration;
88  		this.fileSystem = fileSystem;
89  		this.pathResolver = pathResolver;

103 	/**
104 	 * Execute.
105 	 *
106 	 * @param context the context
107 	 */
108 	@Override
109 	public void execute(final SensorContext context) {
110 		final Profiler profiler = Profiler.create(LOGGER);
111 		profiler.startInfo("Process cfn-nag reports");
112 
113 		try {
114 			final Optional<String> reportFilesProperty = configuration.getReportFiles();
115 
116 			if (reportFilesProperty.isPresent()) {
117 
118 				final String reports = reportFilesProperty.get();
119 				LOGGER.info(CloudformationConstants.REPORT_FILES_PROPERTY + "=" + reports);
120 				final String[] reportFiles = StringUtils.split(reports, ",");
121 
122 				for (final String report : reportFiles) {
123 					LOGGER.info("Processing:" + report);
124 					if (pathResolver.relativeFile(fileSystem.baseDir(), report).exists() && report.endsWith(".nag")) {
125 
126 						handleCfnNagReports(context, report);
127 					} else if (pathResolver.relativeFile(fileSystem.baseDir(), report).exists()
128 							&& report.endsWith(".nagscan")) {
129 
130 						handleCfnNagScanReports(context, report);
131 					} else {
132 						LOGGER.warn("Processing:" + report + " missing or do not end with .nag or .nagscan");
133 					}
134 				}
135 			} else {
136 				LOGGER.warn("Missing property:{}",CloudformationConstants.REPORT_FILES_PROPERTY);
137 			}
138 		} catch (final IOException e) {
139 			throw new RuntimeException("Can not process cfn-nag reports.", e);
140 		} finally {
141 			profiler.stopInfo();
142 		}
143 	}
144 
145 	/**
146 	 * Handle cfn nag reports.
147 	 *
148 	 * @param context the context
149 	 * @param report  the report
150 	 * @throws FileNotFoundException the file not found exception
151 	 */
152 	private void handleCfnNagReports(final SensorContext context, final String report) throws FileNotFoundException {
153 		final String templateName = pathResolver.relativeFile(fileSystem.baseDir(), report).getName().replace(".nag","");
154 
155 		final InputFile templateInputFile = findTemplate(templateName);
156 
157 		final CfnNagReport cfnNagReport = cfnNagReportReader
158 				.readReport(new FileInputStream(pathResolver.relativeFile(fileSystem.baseDir(), report)));
159 		if (cfnNagReport != null) {
160 			final List<CfnNagViolation> violations = cfnNagReport.getViolations();
161 			for (final CfnNagViolation cfnNagViolation : violations) {
162 				addIssue(context, cfnNagViolation, templateInputFile);
163 			}
164 		}
165 	}
166 
167 	/**
168 	 * Handle cfn nag scan reports.
169 	 *
170 	 * @param context the context
171 	 * @param report  the report
172 	 * @throws FileNotFoundException the file not found exception
173 	 */
174 	private void handleCfnNagScanReports(final SensorContext context, final String report)
175 			throws FileNotFoundException {
176 		final List<CfnNagScanReport> cfnNagscanReports = cfnNagScanReportReader
177 				.readReport(new FileInputStream(pathResolver.relativeFile(fileSystem.baseDir(), report)));
178 
179 		for (final CfnNagScanReport nagScanReport : cfnNagscanReports) {
180 
181 			final String filename = nagScanReport.getFilename();
182 			final InputFile templateInputFile = findTemplate(
183 					filename.substring(filename.lastIndexOf(File.separator) + 1, filename.length()));
184 
185 			final List<CfnNagViolation> violations = nagScanReport.getFile_results().getViolations();
186 			for (final CfnNagViolation cfnNagViolation : violations) {
187 				addIssue(context, cfnNagViolation, templateInputFile);
188 			}
189 		}
190 	}
191 
192 	/**
193 	 * Find template.
194 	 *
195 	 * @param templateName the template name
196 	 * @return the input file
197 	 */
198 	private InputFile findTemplate(final String templateName) {
199 		final List<InputFile> potentialReportTargets = new ArrayList<>();
200 		fileSystem.inputFiles(fileSystem.predicates().all()).forEach(potentialReportTargets::add);
201 		LOGGER.info("Looking for cloudformation template matching:" + templateName);
202 
203 		for (final InputFile inputFile : potentialReportTargets) {
204 			if (templateName.equals(inputFile.filename())) {
205 				LOGGER.info("matching:" + templateName + " = " + inputFile.filename());
206 				return inputFile;
207 			}
208 		}
209 		return null;
210 	}

212 	/**
213 	 * Adds the issue.
214 	 *
215 	 * @param context           the context
216 	 * @param violation         the violation
217 	 * @param templateInputFile the template input file
218 	 */
219 	private static void addIssue(final SensorContext context, final CfnNagViolation violation,
220 			final InputFile templateInputFile) {
221 		if (CloudformationQualityProfile.hasRule(violation.getId())) {
222 			if (templateInputFile != null) {
223 
224 				if (violation.getLine_numbers().isEmpty()) {
225 					context.newIssue().forRule(RuleKey.of("cfn-"+templateInputFile.language(), violation.getId()))
226 							.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage()))
227 							.save();
228 				} else {
229 					final List<Integer> line_numbers = violation.getLine_numbers();
230 					for (final Integer line : line_numbers) {
231 						if (line != null && line >= 0) {
232 							context.newIssue()
233 									.forRule(RuleKey.of("cfn-"+templateInputFile.language(), violation.getId()))
234 									.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage())
235 											.at(templateInputFile.selectLine(line)))
236 									.save();
237 						} else {
238 							context.newIssue().forRule(RuleKey.of("cfn-"+templateInputFile.language(), violation.getId()))
239 							.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage()))
240 							.save();
241 						}
242 					}
243 				}
244 			} else {
245 				context.newIssue().forRule(RuleKey.of("cfn-"+"yaml", violation.getId()))
246 						.at(new DefaultIssueLocation().on(context.project()).message(violation.getMessage())).save();
247 			}
248 		} else {
249 			LOGGER.warn("Rule not supported {}:{}", violation.getId(), violation.getMessage());
250 		}
251 	}
