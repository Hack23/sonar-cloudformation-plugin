47  /**
48   * The Class CloudformationSensor.
49   */
50  public class CloudformationSensor implements Sensor {
51  
52  	/** The Constant SENSOR_NAME. */
53  	private static final String SENSOR_NAME = "Cloudformation Check";
54  	
55  	/** The Constant LOGGER. */
56  	private static final Logger LOGGER = Loggers.get(CloudformationSensor.class);
57  
58  	/** The cfn nag report reader. */
59  	private final CfnNagReportReader cfnNagReportReader;
60  	
61  	/** The cfn nag scan report reader. */
62  	private final CfnNagScanReportReader cfnNagScanReportReader;
63  	
64  	/** The configuration. */
65  	private final CloudformationSensorConfiguration configuration;
66  	
67  	/** The path resolver. */
68  	private final PathResolver pathResolver;
69  	
70  	/** The file system. */
71  	private final FileSystem fileSystem;

73  	/**
74  	 * Instantiates a new cloudformation sensor.
75  	 *
76  	 * @param configuration the configuration
77  	 * @param fileSystem the file system
78  	 * @param pathResolver the path resolver
79  	 */
80  	public CloudformationSensor(final CloudformationSensorConfiguration configuration, final FileSystem fileSystem,
81  			final PathResolver pathResolver) {
82  		super();
83  		this.cfnNagReportReader = new CfnNagReportReader();
84  		this.cfnNagScanReportReader = new CfnNagScanReportReader();
85  		this.configuration = configuration;
86  		this.fileSystem = fileSystem;
87  		this.pathResolver = pathResolver;
88  	}

91  	 * Describe.

97  		descriptor.name(SENSOR_NAME);

99  	}

101 	/**
102 	 * Execute.
103 	 *
104 	 * @param context the context
105 	 */
106 	@Override
107 	public void execute(final SensorContext context) {
108 		final Profiler profiler = Profiler.create(LOGGER);
109 		profiler.startInfo("Process cfn-nag reports");
110 
111 		try {
112 			final Optional<String> reportFilesProperty = configuration.getReportFiles();
113 
114 			if (reportFilesProperty.isPresent()) {
115 
116 				final String reports = reportFilesProperty.get();
117 				LOGGER.info(CloudformationConstants.REPORT_FILES_PROPERTY +"=" + reports);
118 				final String[] reportFiles = StringUtils.split(reports, ",");
119 
120 				for (final String report : reportFiles) {
121 					LOGGER.info("Processing:" +report);
122 					if (pathResolver.relativeFile(fileSystem.baseDir(), report).exists() && report.endsWith(".nag")) {
123 
124 						final String templateName = pathResolver.relativeFile(fileSystem.baseDir(), report).getName()
125 								.replace(".nag", "");
126 
127 						final InputFile templateInputFile = findTemplate(templateName);
128 
129 						final CfnNagReport cfnNagReport = cfnNagReportReader.readReport(
130 								new FileInputStream(pathResolver.relativeFile(fileSystem.baseDir(), report)));
131 						if (cfnNagReport != null) {
132 							final List<CfnNagViolation> violations = cfnNagReport.getViolations();
133 							for (final CfnNagViolation cfnNagViolation : violations) {
134 								addIssue(context, cfnNagViolation, templateInputFile);
135 							}
136 						}
137 					} else if (pathResolver.relativeFile(fileSystem.baseDir(), report).exists()
138 							&& report.endsWith(".nagscan")) {
139 
140 						final List<CfnNagScanReport> cfnNagscanReports = cfnNagScanReportReader.readReport(
141 								new FileInputStream(pathResolver.relativeFile(fileSystem.baseDir(), report)));
142 						
143 						for (final CfnNagScanReport nagScanReport : cfnNagscanReports) {
144 														
145 							final String filename = nagScanReport.getFilename();
146 							final InputFile templateInputFile = findTemplate(filename.substring(filename.lastIndexOf(File.separator)+1,filename.length()));
147 							
148 							final List<CfnNagViolation> violations = nagScanReport.getFile_results().getViolations();
149 							for (final CfnNagViolation cfnNagViolation : violations) {
150 								addIssue(context, cfnNagViolation, templateInputFile);
151 							}
152 						}
153 					}
154 				}
155 			}
156 		} catch (final Exception e) {
157 			throw new RuntimeException("Can not process cfn-nag reports.", e);
158 		} finally {
159 			profiler.stopInfo();
160 		}
161 	}
162 
163 	/**
164 	 * Find template.
165 	 *
166 	 * @param templateName the template name
167 	 * @return the input file
168 	 */
169 	private InputFile findTemplate(final String templateName) {
170 		final List<InputFile> potentialReportTargets = new ArrayList<>();
171 		fileSystem.inputFiles(fileSystem.predicates().all()).forEach(potentialReportTargets::add);
172 		LOGGER.info("Looking for cloudformation template matching:" + templateName);
173 
174 		for (final InputFile inputFile : potentialReportTargets) {
175 			if (templateName.equals(inputFile.filename())) {
176 				LOGGER.info("matching:" + templateName + " = " + inputFile.filename());
177 				return inputFile;
178 			}
179 		}
180 		return null;
181 	}
182 
183 	/**
184 	 * Adds the issue.
185 	 *
186 	 * @param context the context
187 	 * @param violation the violation
188 	 * @param templateInputFile the template input file
189 	 */
190 	private void addIssue(final SensorContext context, final CfnNagViolation violation, final InputFile templateInputFile) {
191 		if (templateInputFile != null) {
192 
193 			if (violation.getLine_numbers().isEmpty()) {
194 				context.newIssue().forRule(RuleKey.of(CloudformationRulesDefinition.REPO_KEY, violation.getId()))
195 						.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage())).save();
196 			} else {
197 				final List<Integer> line_numbers = violation.getLine_numbers();
198 				for (final Integer line : line_numbers) {
199 					context.newIssue().forRule(RuleKey.of(CloudformationRulesDefinition.REPO_KEY, violation.getId()))
200 							.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage())
201 									.at(templateInputFile.selectLine(line)))
202 							.save();
203 				}
204 			}
205 		} else {
206 			context.newIssue().forRule(RuleKey.of(CloudformationRulesDefinition.REPO_KEY, violation.getId()))
207 					.at(new DefaultIssueLocation().on(context.project()).message(violation.getMessage())).save();
208 		}
209 	}
