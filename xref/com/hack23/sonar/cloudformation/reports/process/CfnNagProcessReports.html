42  /**
43   * The Class CfnNagProcessReports.
44   */
45  public final class CfnNagProcessReports extends AbstractProcessReports {
46  
47  	/** The Constant UNDEFINED_FAILURE. */
48  	public static final String UNDEFINED_FAILURE = "FUNDEFINED";
49  
50  	/** The Constant UNDEFINED_WARNING. */
51  	public static final String UNDEFINED_WARNING = "WUNDEFINED";
52  
53  	/** The Constant LOGGER. */
54  	private static final Logger LOGGER = Loggers.get(CfnNagProcessReports.class);
55  
56  	/** The cfn nag scan report reader. */
57  	private final CfnNagScanReportReader cfnNagScanReportReader;
58  
59  	/** The path resolver. */
60  	private final PathResolver pathResolver;
61  
62  	/** The file system. */
63  	private final FileSystem fileSystem;
64  
65  
66  	/**
67  	 * Instantiates a new cfn nag process reports.
68  	 *
69  	 * @param fileSystem the file system
70  	 * @param pathResolver the path resolver
71  	 */
72  	public CfnNagProcessReports(final FileSystem fileSystem, final PathResolver pathResolver) {
73  		super();
74  		this.cfnNagScanReportReader = new CfnNagScanReportReader();
75  		this.fileSystem = fileSystem;
76  		this.pathResolver = pathResolver;
77  	}
78  
79  	/**
80  	 * Process cfn nag report.
81  	 *
82  	 * @param context the context
83  	 * @param reportFilesProperty the report files property
84  	 * @throws IOException Signals that an I/O exception has occurred.
85  	 */
86  	public void processCfnNagReport(final SensorContext context, final Optional<String> reportFilesProperty)
87  			throws IOException {
88  		if (reportFilesProperty.isPresent()) {
89  
90  			final String reports = reportFilesProperty.get();
91  			final String[] reportFiles = StringUtils.split(reports, ",");
92  
93  			for (final String report : reportFiles) {
94  				LOGGER.info("Processing cfn-nag :" + report);
95  				if (pathResolver.relativeFile(fileSystem.baseDir(), report).exists()) {
96  
97  					handleCfnNagScanReports(context, report);
98  				} else {
99  					LOGGER.warn("Processing cfn-nag:" + report + " missing");
100 				}
101 			}
102 		} else {
103 			LOGGER.warn("Missing property:{}", CloudformationConstants.CFN_NAG_REPORT_FILES_PROPERTY);
104 		}
105 	}
106 
107 	/**
108 	 * Handle cfn nag scan reports.
109 	 *
110 	 * @param context the context
111 	 * @param report the report
112 	 * @throws IOException 
113 	 */
114 	private void handleCfnNagScanReports(final SensorContext context, final String report)
115 			throws IOException {
116 		LOGGER.info("Reading cfn-nag reports:{}", report);
117 		final List<CfnNagScanReport> cfnNagscanReports = cfnNagScanReportReader
118 				.readReport(Files.newInputStream(pathResolver.relativeFile(fileSystem.baseDir(), report).toPath()));
119 
120 		for (final CfnNagScanReport nagScanReport : cfnNagscanReports) {
121 
122 			final String filename = nagScanReport.getFilename();
123 			LOGGER.info("Cfn-nag scanned file :{}", filename);
124 
125 			final InputFile templateInputFile = findTemplate(fileSystem,
126 					filename.substring(filename.lastIndexOf(File.separator) + 1, filename.length()), filename);
127 
128 			final List<CfnNagViolation> violations = nagScanReport.getFileResults().getViolations();
129 			for (final CfnNagViolation cfnNagViolation : violations) {
130 				addIssue(context, cfnNagViolation, templateInputFile);
131 			}
132 		}
133 	}
134 
135 	/**
136 	 * Adds the issue.
137 	 *
138 	 * @param context the context
139 	 * @param violation the violation
140 	 * @param templateInputFile the template input file
141 	 */
142 	private static void addIssue(final SensorContext context, final CfnNagViolation violation,
143 			final InputFile templateInputFile) {
144 		final ActiveRules activeRules = context.activeRules();
145 
146 		if (templateInputFile != null) {
147 
148 			if (violation.getLineNumbers().isEmpty()) {
149 				context.newIssue().forRule(RuleKey.of("cfn-" + templateInputFile.language(), findRuleId(activeRules, violation)))
150 						.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage())).save();
151 			} else {
152 				final List<Integer> lineNumbers = violation.getLineNumbers();
153 				for (final Integer line : lineNumbers) {
154 					if (line != null && line >= 0) {
155 						context.newIssue()
156 								.forRule(RuleKey.of("cfn-" + templateInputFile.language(), findRuleId(activeRules, violation)))
157 								.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage())
158 										.at(templateInputFile.selectLine(line)))
159 								.save();
160 					} else {
161 						context.newIssue()
162 								.forRule(RuleKey.of("cfn-" + templateInputFile.language(), findRuleId(activeRules, violation)))
163 								.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage()))
164 								.save();
165 					}
166 				}
167 			}
168 		} else {
169 			context.newIssue().forRule(RuleKey.of("cfn-yaml", findRuleId(activeRules, violation)))
170 					.at(new DefaultIssueLocation().on(context.project()).message(violation.getMessage())).save();
171 		}
172 	}
173 
174 	/**
175 	 * Find rule id.
176 	 *
177 	 * @param violation the violation
178 	 * @return the string
179 	 */
180 	private static String findRuleId(final ActiveRules activeRules, final CfnNagViolation violation) {
181 		RuleKey ruleKeyYaml = RuleKey.of("cfn-yaml", violation.getId());
182 		RuleKey ruleKeyJson = RuleKey.of("cfn-json", violation.getId());
183 		
184 		if (activeRules.find(ruleKeyYaml) != null || activeRules.find(ruleKeyJson) != null) {
185 			return violation.getId();
186 		} else {
187 			if (violation.getId().startsWith("W")) {
188 				return UNDEFINED_WARNING;
189 			} else {
190 				return UNDEFINED_FAILURE;
191 			}
192 		}
193 	}
