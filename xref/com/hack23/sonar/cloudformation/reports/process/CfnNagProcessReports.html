267 	/**
268 	 * Process cfn nag report.
269 	 *
270 	 * @param context the context
271 	 * @param reportFilesProperty the report files property
272 	 * @throws IOException Signals that an I/O exception has occurred.
273 	 */
274 	public void processCfnNagReport(final SensorContext context, final Optional<String> reportFilesProperty)
275 			throws IOException {
276 		if (reportFilesProperty.isPresent()) {
277 
278 			final String reports = reportFilesProperty.get();
279 			final String[] reportFiles = StringUtils.split(reports, ",");
280 
281 			for (final String report : reportFiles) {
282 				LOGGER.info("Processing cfn-nag :" + report);
283 				if (pathResolver.relativeFile(fileSystem.baseDir(), report).exists()) {
284 
285 					handleCfnNagScanReports(context, report);
286 				} else {
287 					LOGGER.warn("Processing cfn-nag:" + report + " missing");
288 				}
289 			}
290 		} else {
291 			LOGGER.warn("Missing property:{}", CloudformationConstants.CFN_NAG_REPORT_FILES_PROPERTY);
292 		}
293 	}

295 	/**
296 	 * Handle cfn nag scan reports.
297 	 *
298 	 * @param context the context
299 	 * @param report the report
300 	 * @throws IOException 
301 	 */
302 	private void handleCfnNagScanReports(final SensorContext context, final String report)
303 			throws IOException {
304 		LOGGER.info("Reading cfn-nag reports:{}", report);
305 		final List<CfnNagScanReport> cfnNagscanReports = cfnNagScanReportReader
306 				.readReport(Files.newInputStream(pathResolver.relativeFile(fileSystem.baseDir(), report).toPath()));
307 
308 		for (final CfnNagScanReport nagScanReport : cfnNagscanReports) {
309 
310 			final String filename = nagScanReport.getFilename();
311 			LOGGER.info("Cfn-nag scanned file :{}", filename);
312 
313 			final InputFile templateInputFile = findTemplate(fileSystem,
314 					filename.substring(filename.lastIndexOf(File.separator) + 1, filename.length()), filename);
315 
316 			final List<CfnNagViolation> violations = nagScanReport.getFileResults().getViolations();
317 			for (final CfnNagViolation cfnNagViolation : violations) {
318 				addIssue(context, cfnNagViolation, templateInputFile);
319 			}
320 		}
321 	}

323 	/**
324 	 * Adds the issue.
325 	 *
326 	 * @param context the context
327 	 * @param violation the violation
328 	 * @param templateInputFile the template input file
329 	 */
330 	private static void addIssue(final SensorContext context, final CfnNagViolation violation,
331 			final InputFile templateInputFile) {
332 		if (templateInputFile != null) {
333 
334 			if (violation.getLineNumbers().isEmpty()) {
335 				context.newIssue().forRule(RuleKey.of("cfn-" + templateInputFile.language(), findRuleId(violation)))
336 						.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage())).save();
337 			} else {
338 				final List<Integer> lineNumbers = violation.getLineNumbers();
339 				for (final Integer line : lineNumbers) {
340 					if (line != null && line >= 0) {
341 						context.newIssue()
342 								.forRule(RuleKey.of("cfn-" + templateInputFile.language(), findRuleId(violation)))
343 								.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage())
344 										.at(templateInputFile.selectLine(line)))
345 								.save();
346 					} else {
347 						context.newIssue()
348 								.forRule(RuleKey.of("cfn-" + templateInputFile.language(), findRuleId(violation)))
349 								.at(new DefaultIssueLocation().on(templateInputFile).message(violation.getMessage()))
350 								.save();
351 					}
352 				}
353 			}
354 		} else {
355 			context.newIssue().forRule(RuleKey.of("cfn-yaml", findRuleId(violation)))
356 					.at(new DefaultIssueLocation().on(context.project()).message(violation.getMessage())).save();
357 		}
358 	}

360 	/**
361 	 * Find rule id.
362 	 *
363 	 * @param violation the violation
364 	 * @return the string
365 	 */
366 	private static String findRuleId(final CfnNagViolation violation) {
367 		if (hasRule(violation.getId())) {
368 			return violation.getId();
369 		} else {
370 			if (violation.getId().startsWith("W")) {
371 				return UNDEFINED_WARNING;
372 			} else {
373 				return UNDEFINED_FAILURE;
374 			}
375 		}
