139 	public static boolean hasRule(final String id) {
140 		return SUPPORTED_RULES.contains(id);
141 	}
142 	
143 	/**
144 	 * Define.
145 	 *
146 	 * @param context the context
147 	 */
148 	@Override
149 	public void define(final Context context) {
150 		final NewBuiltInQualityProfile profile = context.createBuiltInQualityProfile("Cloudformation Rules",
151 				CloudformationLanguage.KEY);
152 		profile.setDefault(true);
153 
154 		for (final String ruleKey : SUPPORTED_RULES) {
155 			profile.activateRule(CloudformationRulesDefinition.REPO_KEY, ruleKey);
156 		}
157 
158 		profile.done();
159 	}
