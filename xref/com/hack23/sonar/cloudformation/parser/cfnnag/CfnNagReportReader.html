20  package com.hack23.sonar.cloudformation.parser.cfnnag;
21  
22  import java.io.IOException;
23  import java.io.InputStream;
24  
25  import org.sonar.api.utils.log.Logger;
26  import org.sonar.api.utils.log.Loggers;
27  
28  import com.fasterxml.jackson.databind.ObjectMapper;
29  
30  // TODO: Auto-generated Javadoc
31  /**
32   * The Class CfnNagReportReader.
33   */
34  public class CfnNagReportReader {
35  
36  	/** The Constant LOGGER. */
37  	private static final Logger LOGGER = Loggers.get(CfnNagReportReader.class);
38  
39  	/**
40  	 * Read report.
41  	 *
42  	 * @param input the input
43  	 * @return the cfn nag report
44  	 */
45  	public CfnNagReport readReport(final InputStream input) {
46  		final ObjectMapper objectMapper = new ObjectMapper();
47  
48  		try {
49  			return objectMapper.readValue(input, CfnNagReport.class);
50  		} catch (final IOException e) {
51  			LOGGER.warn("Problem reading cfn nag report json:{}", e.getMessage());
52  			return new CfnNagReport();
53  		}
54  	}
55  }
