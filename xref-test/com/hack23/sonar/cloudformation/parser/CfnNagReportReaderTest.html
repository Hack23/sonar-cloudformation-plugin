20  package com.hack23.sonar.cloudformation.parser;
21  
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * The Class CfnNagReportReaderTest.
27   */
28  public class CfnNagReportReaderTest extends Assert {
29  
30  	/**
31  	 * Read report test.
32  	 */
33  	@Test
34  	public void readReportTest() {
35  		final CfnNagReport cfnNagReport = new CfnNagReportReader().readReport(CfnNagReportReaderTest.class.getResourceAsStream("/aws-cross-account-manager-master.yml.nag"));
36  		assertNotNull(cfnNagReport);
37  		assertEquals(1,cfnNagReport.getFailure_count());
38  		assertEquals(5,cfnNagReport.getViolations().size());
39  		final CfnNagViolation nagViolation = cfnNagReport.getViolations().get(0);
40  		assertEquals("W12",nagViolation.getId());
41  		assertEquals("WARN",nagViolation.getType());
42  		assertEquals("IAM policy should not allow * resource",nagViolation.getMessage());
43  		assertEquals(4,nagViolation.getLogical_resource_ids().size());
44  		assertEquals(4,nagViolation.getLine_numbers().size());
45  	}
46  }
