53   * The Class ModelSanityTest.
54   */
55  public final class ModelSanityTest extends Assert {
56  
57  	private static final FilterPackageInfo FilterPackageInfo = new FilterPackageInfo();
58  
59  	/** The Constant EXPECT_CLASSES_IN_PACKAGE. */
60  	private static final String EXPECT_CLASSES_IN_PACKAGE = "Expect classes in package";
61  
62  	
63  	@Test
64  	public void modelTest() {
65  
66  		assertTrue(EXPECT_CLASSES_IN_PACKAGE,
67  				checkAllClassesInPackage( ModelSanityTest.class.getPackage().getName()));
68  
69  	}
70  
71  	protected final boolean checkAllClassesInPackage(final String string) {
72  		final List<PojoClass> pojoClassesRecursively = PojoClassFactory.getPojoClassesRecursively(string,
73  				new FilterTestClasses());
74  
75  		final Validator validator = ValidatorBuilder.create().with(new SetterMustExistRule(), new GetterMustExistRule())
76  				.with(new SetterTester(), new GetterTester()).with(new InvokeToStringTester())
77  				.with(new InvokeHashcodeTester()).with(new DummyEqualsTester()).with(new EqualsAndHashCodeMatchRule()).build();
78  		validator.validate(pojoClassesRecursively);
79  
80  
81  		return true;
82  	}
83  
84  	private static class FilterTestClasses implements PojoClassFilter {
85  		public boolean include(final PojoClass pojoClass) {
86  			return !(pojoClass.getSourcePath().contains("/test-classes/")
87  					|| pojoClass.getClazz().getName().contains("_") || pojoClass.isEnum() || pojoClass.isAbstract())
88  					&& FilterPackageInfo.include(pojoClass);
89  		}
90  	}
91  
92  	private static class InvokeToStringTester implements Tester {
93  		public void run(final PojoClass pojoClass) {
94  			final Object instance = RandomFactory.getRandomValue(pojoClass.getClazz());
95  			Affirm.affirmNotNull("toStringFailure", instance.toString());
96  		}
97  	}
98  
99  	/**
100 	 * The Class InvokeHashcodeTester.
101 	 */
102 	private static class InvokeHashcodeTester implements Tester {
103 		public void run(final PojoClass pojoClass) {
104 			final Object instance = RandomFactory.getRandomValue(pojoClass.getClazz());
105 			Affirm.affirmFalse("hashCodeFailure", 0 == instance.hashCode());
106 		}
107 	}
108 
109 	/**
110 	 * The Class DummyEqualsTester.
111 	 */
112 	private static class DummyEqualsTester implements Tester {
113 		public void run(final PojoClass pojoClass) {
114 			final Object instance = randomValues(pojoClass);
115 
116 			Affirm.affirmFalse("EqualsCompareNullFailure", instance.equals(null));
117 			Affirm.affirmFalse("EqualsCompareWrongClassFailure", instance.equals("WrongClass"));
118 			Affirm.affirmTrue("EqualsCompareSelfFailure", instance.equals(instance));			
119 			
120 			final Object instance2 = randomValues(pojoClass);
121 
122 			instance.equals(instance2);
123 		}
124 
125 		private Object randomValues(final PojoClass pojoClass) {
126 			final Object instance = RandomFactory.getRandomValue(pojoClass.getClazz());
127 			randomValues(instance, pojoClass);
128 
129 			return instance;
130 		}
131 
132 		private static void randomValues(final Object instance, final PojoClass pojoClass) {
133 			if (pojoClass == null) {
134 				return;
135 			}
136 
137 			for (final PojoField fieldEntry : pojoClass.getPojoFields()) {
138 				if (fieldEntry.hasSetter()) {
139 					final Object value;
140 
141 					value = RandomFactory.getRandomValue(fieldEntry);
142 					fieldEntry.invokeSetter(instance, value);
143 				}
144 			}
145 			randomValues(instance, pojoClass.getSuperClass());
146 		}
147 	}
