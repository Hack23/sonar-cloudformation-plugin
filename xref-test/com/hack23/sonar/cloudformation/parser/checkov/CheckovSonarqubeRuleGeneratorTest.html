113 	public void generateSonarqubeRuleDefinitionsForCheckovTest() throws IOException {
114 		final CSVParser parser = CSVParser.parse(new InputStreamReader(this.getClass().getResourceAsStream("/checkov/rules.txt"),StandardCharsets.UTF_8), CSVFormat.EXCEL.withHeader().withDelimiter('|').withTrim());
115 		final List<CSVRecord> records = parser.getRecords();
116 		records.remove(0);
117 		Map<String,String> map = new HashMap<>();
118 		for (final CSVRecord csvRecord : records) {
119 			if (csvRecord.isSet("Id") && "resource".equals(csvRecord.get("Type")) && !map.containsKey(csvRecord.get("IaC") +"-" + csvRecord.get("Id"))) {
120 				String ruleEntryUntagged = XML_ENTRY.replace("{RULE_ID}",csvRecord.get("Id")).replace("{NAME}",csvRecord.get("Policy")).replace("{IaC}",csvRecord.get("IaC").toLowerCase()).replace("\"","&quot;");
121 				
122 				for (String key : NIST_POLICY_STRING_MAPPING.keySet()) {
123 					if (ruleEntryUntagged.toLowerCase().contains(key)) {
124 						ruleEntryUntagged = ruleEntryUntagged.replace("{EXTRA_TAGS}",NIST_POLICY_STRING_MAPPING.get(key));
125 					}
126 				}
127 				ruleEntryUntagged = ruleEntryUntagged.replace("{EXTRA_TAGS}","");
128 				System.out.println(ruleEntryUntagged);
129 				
130 				map.put(csvRecord.get("IaC") +"-" + csvRecord.get("Id"),"");
131 			}
132 		}
